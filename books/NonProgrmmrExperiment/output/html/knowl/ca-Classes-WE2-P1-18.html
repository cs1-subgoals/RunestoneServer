public class Book {
    private String title;
    private String author;
    private String publisher;
    private int copyrightDate;
    public Book() {
        title = "";
        author = "";
        publisher = "";
        copyrightDate = -1;
    }

    public Book(String t, String a, String p, int c) {
        title = t;
        author = a;
        publisher = p;
        copyrightDate = c;
    }
}
