    Q5: What is printed as a result of executing the following code segment?
    int typeA = 0;
int typeB = 0;
int typeC = 0;
for (int k = 1; k <= 50; k++) {
   if (k % 2 == 0 && k % 5 == 0)
      typeA++;
   if (k % 2 == 0)
      typeB++;
   if (k % 5 == 0)
      typeC++;
}
System.out.println("%d %d %d", typeA, typeB, typeC);
